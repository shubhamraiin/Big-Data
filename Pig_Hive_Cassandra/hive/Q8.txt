DROP TABLE IF EXISTS MOVIES_JAN;
CREATE TABLE MOVIES_JAN (MOVIEID STRING,TITLE STRING,GENRES STRING) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\;';

DROP TABLE IF EXISTS MOVIES_FEB;
CREATE TABLE MOVIES_FEB (MOVIEID STRING,TITLE STRING,GENRES STRING) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\;';

DROP TABLE IF EXISTS MOVIES_MAR;
CREATE TABLE MOVIES_MAR (MOVIEID STRING,TITLE STRING,GENRES STRING) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\;';

FROM PARTITION_MOVIES
INSERT OVERWRITE TABLE MOVIES_JAN SELECT PARTITION_MOVIES.MOVIEID, PARTITION_MOVIES.TITLES, PARTITION_MOVIES.GENRES WHERE PARTITION_MOVIES.MONTH = "January"
INSERT OVERWRITE TABLE MOVIES_FEB SELECT PARTITION_MOVIES.MOVIEID, PARTITION_MOVIES.TITLES, PARTITION_MOVIES.GENRES WHERE PARTITION_MOVIES.MONTH = "February"
INSERT OVERWRITE TABLE MOVIES_MAR SELECT PARTITION_MOVIES.MOVIEID, PARTITION_MOVIES.TITLES, PARTITION_MOVIES.GENRES WHERE PARTITION_MOVIES.MONTH = "March";

SELECT * FROM MOVIES_JAN;
SELECT * FROM MOVIES_FEB;
